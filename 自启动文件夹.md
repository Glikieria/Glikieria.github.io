在Windows系统中，有两个自启动文件夹，第一个是用户个人启动文件夹，位于
C:\Users\USERNAME\AppData\Roaming\Microsoft\Windows\StartMenu\Programs\Startup
其中USERNAME是用户名。
另一个自启动文件夹是公共启动文件夹，包含为计算机上的每个用户自动运行的程序，位于
C:\ProgramData\Microsoft\Windows\Start Menu\Programs\Startup。
```
#include <windows.h>
#include <shlobj.h>     // 用于 SHGetKnownFolderPath
#include <iostream>
#include <string>

#pragma comment(lib, "shell32.lib")

// 获取当前可执行文件的完整路径
std::wstring GetCurrentExePath() {
	wchar_t path[MAX_PATH];
	GetModuleFileName(NULL, path, MAX_PATH);
	return std::wstring(path);
}

// 获取指定用户的启动文件夹路径
// 参数: CSIDL_STARTUP  (个人启动文件夹)
//      CSIDL_COMMON_STARTUP (公共启动文件夹)
std::wstring GetStartupFolder(int csidl) {
	wchar_t path[MAX_PATH];
	if (SUCCEEDED(SHGetFolderPath(NULL, csidl, NULL, SHGFP_TYPE_CURRENT, path))) {
		return std::wstring(path);
	}
	return L"";
}

int main() {
	// 获取当前程序路径
	std::wstring exePath = GetCurrentExePath();
	if (exePath.empty()) {
		std::wcerr << L"获取当前程序路径失败" << std::endl;
		return 1;
	}

	// 获取个人启动文件夹（当前用户）
	std::wstring userStartup = GetStartupFolder(CSIDL_STARTUP);
	if (!userStartup.empty()) {
		std::wstring destPath = userStartup + L"\\" +
			exePath.substr(exePath.find_last_of(L'\\') + 1);
		if (CopyFile(exePath.c_str(), destPath.c_str(), FALSE)) {
			std::wcout << L"已添加到个人启动文件夹: " << destPath << std::endl;
		}
		else {
			std::wcerr << L"复制到个人启动文件夹失败，错误码: " << GetLastError() << std::endl;
		}
	}
	else {
		std::wcerr << L"获取个人启动文件夹失败" << std::endl;
	}

	// 获取公共启动文件夹（所有用户）—— 通常需要管理员权限才能写入
	std::wstring commonStartup = GetStartupFolder(CSIDL_COMMON_STARTUP);
	if (!commonStartup.empty()) {
		std::wstring destPath = commonStartup + L"\\" +
			exePath.substr(exePath.find_last_of(L'\\') + 1);
		if (CopyFile(exePath.c_str(), destPath.c_str(), FALSE)) {
			std::wcout << L"已添加到公共启动文件夹: " << destPath << std::endl;
		}
		else {
			std::wcerr << L"复制到公共启动文件夹失败，错误码: " << GetLastError() << std::endl;
		}
	}
	else {
		std::wcerr << L"获取公共启动文件夹失败" << std::endl;
	}

	return 0;
}
```
