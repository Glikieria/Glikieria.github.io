一个隐蔽的启动项:
HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Session Manager\Environment\UserInitMprLogonScript在Environment下创建一个userinitmprlogonscript的建类型为reg_sz值就填要自启动的应用程序路径

```

/**
 * 设置系统级登录脚本（UserInitMprLogonScript）
 * 路径: HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Session Manager\Environment
 * 值名: UserInitMprLogonScript (REG_SZ)
 * @param scriptPath 要执行的脚本/程序的完整路径（宽字符字符串）
 * @return 成功返回 true，失败返回 false
 * @note 此操作需要管理员权限，否则会因权限不足而失败
 */
bool SetSystemUserInitMprLogonScript(const std::wstring& scriptPath) {
    HKEY hKey = NULL;
    LPCWSTR subKey = L"SYSTEM\\CurrentControlSet\\Control\\Session Manager\\Environment";

    // 尝试打开注册表键（需要写入权限）
    LONG result = RegOpenKeyExW(HKEY_LOCAL_MACHINE, subKey, 0, KEY_SET_VALUE, &hKey);
    if (result != ERROR_SUCCESS) {
        // 键不存在，则创建（赋予写入权限）
        result = RegCreateKeyExW(HKEY_LOCAL_MACHINE, subKey, 0, NULL,
            REG_OPTION_NON_VOLATILE, KEY_SET_VALUE,
            NULL, &hKey, NULL);
        if (result != ERROR_SUCCESS) {
            return false;   // 打开/创建失败
        }
    }

    // 设置值：名称为 UserInitMprLogonScript，类型为 REG_SZ
    DWORD dataSize = static_cast<DWORD>((scriptPath.size() + 1) * sizeof(wchar_t));
    result = RegSetValueExW(hKey, L"UserInitMprLogonScript", 0, REG_SZ,
        reinterpret_cast<const BYTE*>(scriptPath.c_str()),
        dataSize);
    RegCloseKey(hKey);
    return (result == ERROR_SUCCESS);
}
```
