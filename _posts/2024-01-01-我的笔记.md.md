//写入自启动
DWORD AutoRun()
{
    HKEY hKey;
    // 1. 打开注册表启动项路径
    if (RegOpenKeyEx(HKEY_CURRENT_USER,
                     _T("SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run"),
                     0,
                     KEY_ALL_ACCESS,
                     &hKey) == ERROR_SUCCESS)
    {
        // 2. 获取当前程序的完整路径
        TCHAR szExePath[MAX_PATH];
        GetModuleFileName(NULL, szExePath, MAX_PATH);

        // 3. 检查是否已存在同名启动项
        TCHAR szExistingPath[MAX_PATH] = {};
        DWORD dwSize = MAX_PATH;
        LONG lResult = RegGetValue(hKey,nullptr, _T("MyApp"),RRF_RT_REG_SZ, 0,
                                   szExistingPath,
                                   &dwSize);

        // 4. 若不存在或路径不一致，则添加/更新启动项
        if (lResult != ERROR_SUCCESS || _tcscmp(szExePath, szExistingPath) != 0)
        {
            RegSetValueEx(hKey,
                          _T("MyApp"),
                          0,
                          REG_SZ,
                          (LPBYTE)szExePath,
                          (lstrlen(szExePath) + 1) * sizeof(TCHAR));
        }

        // 5. 关闭注册表句柄
        RegCloseKey(hKey);
    }

    return 0;
}
//取消自启动
void CancelAutoRun()
{
    HKEY hKey;
    // 1. 打开注册表启动项路径
    if (RegOpenKeyEx(HKEY_CURRENT_USER,
                     _T("SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run"),
                     0,
                     KEY_ALL_ACCESS,
                     &hKey) == ERROR_SUCCESS)
    {
        // 2. 删除指定的启动项
        RegDeleteValue(hKey, _T("MyApp"));

        // 3. 关闭注册表句柄
        RegCloseKey(hKey);
    }
}
