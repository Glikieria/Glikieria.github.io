```
task.h

#pragma once
#include <taskschd.h>
#pragma comment(lib, "taskschd.lib")

class CMyTaskSchedule
{
public:
    CMyTaskSchedule();
    ~CMyTaskSchedule();

    // 创建计划任务
    // lpszTaskName    : 任务名称
    // lpszProgramPath : 要执行的程序路径
    // lpszParameters  : 执行参数（可为 NULL）
    // lpszAuthor      : 作者信息
    BOOL NewTask(char* lpszTaskName, char* lpszProgramPath, char* lpszParameters, char* lpszAuthor);

    // 删除计划任务
    // lpszTaskName : 任务名称
    BOOL Delete(char* lpszTaskName);

private:
    ITaskService* m_lpITS;       // 任务服务接口
    ITaskFolder*  m_lpRootFolder; // 根任务文件夹接口
};

```
task.cpp
#include "task.h"
#include <atlcomcli.h>
#include <comutil.h>
#pragma comment(lib, "comsuppw.lib")

// 构造函数：初始化 COM 环境并连接任务计划服务
CMyTaskSchedule::CMyTaskSchedule()
    : m_lpITS(NULL)
    , m_lpRootFolder(NULL)
{
    // 初始化 COM 库
    HRESULT hr = CoInitialize(NULL);
    if (FAILED(hr))
    {
        MessageBox(NULL, L"初始化COM接口环境失败!", L"Error", MB_OK);
        return;
    }

    // 创建任务计划服务对象
    hr = CoCreateInstance(CLSID_TaskScheduler, NULL,
                          CLSCTX_INPROC_SERVER, IID_ITaskService,
                          (LPVOID*)&m_lpITS);
    if (FAILED(hr))
    {
        MessageBox(NULL, L"创建任务服务对象失败!", L"Error", MB_OK);
        CoUninitialize();   // 确保 COM 被正确卸载
        return;
    }

    // 连接到本地任务计划服务
    hr = m_lpITS->Connect(_variant_t(), _variant_t(), _variant_t(), _variant_t());
    if (FAILED(hr))
    {
        MessageBox(NULL, L"连接到任务服务失败!", L"Error", MB_OK);
        m_lpITS->Release();
        m_lpITS = NULL;
        CoUninitialize();
        return;
    }

    // 获取根任务文件夹（即 "\"）
    hr = m_lpITS->GetFolder(_bstr_t("\\"), &m_lpRootFolder);
    if (FAILED(hr))
    {
        MessageBox(NULL, L"获取根任务的指针失败!", L"Error", MB_OK);
        m_lpITS->Release();
        m_lpITS = NULL;
        CoUninitialize();
    }
}

// 析构函数：释放接口资源，卸载 COM
CMyTaskSchedule::~CMyTaskSchedule()
{
    if (m_lpRootFolder)
        m_lpRootFolder->Release();
    if (m_lpITS)
        m_lpITS->Release();
    CoUninitialize();
}

// 创建计划任务
BOOL CMyTaskSchedule::NewTask(char* lpszTaskName, char* lpszProgramPath,
                              char* lpszParameters, char* lpszAuthor)
{
    if (!m_lpRootFolder)
        return FALSE;

    // 若任务已存在，先删除
    Delete(lpszTaskName);

    // 创建任务定义对象
    ITaskDefinition* pTaskDefinition = NULL;
    HRESULT hr = m_lpITS->NewTask(0, &pTaskDefinition);
    if (FAILED(hr))
    {
        MessageBox(NULL, L"创建任务失败!", L"Error", MB_OK);
        return FALSE;
    }

    // -------------------- 设置注册信息（作者等） --------------------
    IRegistrationInfo* pRegInfo = NULL;
    hr = pTaskDefinition->get_RegistrationInfo(&pRegInfo);
    if (SUCCEEDED(hr))
    {
        // 设置作者
        CComVariant variantAuthor(lpszAuthor);
        pRegInfo->put_Author(variantAuthor.bstrVal);
        pRegInfo->Release();
    }
    else
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"设置注册信息失败!", L"Error", MB_OK);
        return FALSE;
    }

    // -------------------- 设置主体（登录类型、运行权限） --------------------
    IPrincipal* pPrincipal = NULL;
    hr = pTaskDefinition->get_Principal(&pPrincipal);
    if (SUCCEEDED(hr))
    {
        // 使用交互式登录（用户登录时运行）
        pPrincipal->put_LogonType(TASK_LOGON_INTERACTIVE_TOKEN);
        // 使用最高权限运行
        pPrincipal->put_RunLevel(TASK_RUNLEVEL_HIGHEST);
        pPrincipal->Release();
    }
    else
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"设置登录类型和运行权限失败!", L"Error", MB_OK);
        return FALSE;
    }

    // -------------------- 设置其他任务属性 --------------------
    ITaskSettings* pSettings = NULL;
    hr = pTaskDefinition->get_Settings(&pSettings);
    if (SUCCEEDED(hr))
    {
        // 允许在电池模式下运行
        pSettings->put_StopIfGoingOnBatteries(VARIANT_FALSE);
        pSettings->put_DisallowStartIfOnBatteries(VARIANT_FALSE);
        // 允许手动启动
        pSettings->put_AllowDemandStart(VARIANT_TRUE);
        // 不强制在可用时立即启动
        pSettings->put_StartWhenAvailable(VARIANT_FALSE);
        // 允许多个实例并行运行
        pSettings->put_MultipleInstances(TASK_INSTANCES_PARALLEL);
        pSettings->Release();
    }
    else
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"设置其他信息失败!", L"Error", MB_OK);
        return FALSE;
    }

    // -------------------- 创建执行动作（运行程序） --------------------
    IActionCollection* pActionCollect = NULL;
    hr = pTaskDefinition->get_Actions(&pActionCollect);
    if (FAILED(hr))
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"创建执行动作失败!", L"Error", MB_OK);
        return FALSE;
    }

    IAction* pAction = NULL;
    hr = pActionCollect->Create(TASK_ACTION_EXEC, &pAction);
    pActionCollect->Release();  // 释放集合，后续通过 pAction 操作
    if (FAILED(hr))
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"创建执行操作失败!", L"Error", MB_OK);
        return FALSE;
    }

    // 查询 IExecAction 接口，用于设置路径和参数
    IExecAction* pExecAction = NULL;
    hr = pAction->QueryInterface(IID_IExecAction, (LPVOID*)&pExecAction);
    pAction->Release();  // 释放 IAction 接口
    if (FAILED(hr))
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"获取执行动作接口失败!", L"Error", MB_OK);
        return FALSE;
    }

    // 设置程序路径和参数
    CComVariant variantProgramPath(lpszProgramPath);
    pExecAction->put_Path(variantProgramPath.bstrVal);
    if (lpszParameters)
    {
        CComVariant variantParameters(lpszParameters);
        pExecAction->put_Arguments(variantParameters.bstrVal);
    }
    pExecAction->Release();

    // -------------------- 创建触发器（用户登录时触发） --------------------
    ITriggerCollection* pTriggers = NULL;
    hr = pTaskDefinition->get_Triggers(&pTriggers);
    if (SUCCEEDED(hr))
    {
        ITrigger* pTrigger = NULL;
        hr = pTriggers->Create(TASK_TRIGGER_LOGON, &pTrigger);
        pTriggers->Release();  // 释放集合
        if (FAILED(hr))
        {
            pTaskDefinition->Release();
            MessageBox(NULL, L"创建触发器失败!", L"Error", MB_OK);
            return FALSE;
        }
        // 触发器无需额外设置即可使用（默认在任意用户登录时触发）
        pTrigger->Release();
    }
    else
    {
        pTaskDefinition->Release();
        MessageBox(NULL, L"设置触发器信息失败!", L"Error", MB_OK);
        return FALSE;
    }

    // -------------------- 注册任务到计划任务文件夹 --------------------
    IRegisteredTask* pRegisteredTask = NULL;
    CComVariant variantTaskName(lpszTaskName);
    hr = m_lpRootFolder->RegisterTaskDefinition(
        variantTaskName.bstrVal,
        pTaskDefinition,
        TASK_CREATE_OR_UPDATE,          // 若存在则更新
        _variant_t(),                    // 默认用户
        _variant_t(),                    // 默认密码
        TASK_LOGON_INTERACTIVE_TOKEN,    // 交互式登录
        _variant_t(""),                  // 无额外SDDL
        &pRegisteredTask
    );
    pTaskDefinition->Release();  // 任务定义已使用完毕

    if (FAILED(hr))
    {
        MessageBox(NULL, L"注册任务计划失败!", L"Error", MB_OK);
        return FALSE;
    }

    pRegisteredTask->Release();
    return TRUE;
}

// 删除计划任务
BOOL CMyTaskSchedule::Delete(char* lpszTaskName)
{
    if (!m_lpRootFolder)
        return FALSE;

    CComVariant variantTaskName(lpszTaskName);
    HRESULT hr = m_lpRootFolder->DeleteTask(variantTaskName.bstrVal, 0);
    return SUCCEEDED(hr);
}
